# 데이터베이스 설계 (Supabase/PostgreSQL)

## 1. ERD 개요

```
┌─────────────┐       ┌─────────────┐       ┌─────────────┐
│   games     │       │   prizes    │       │   draws     │
├─────────────┤       ├─────────────┤       ├─────────────┤
│ id (PK)     │──┐    │ id (PK)     │──┐    │ id (PK)     │
│ name        │  │    │ game_id(FK) │◄─┤    │ prize_id(FK)│◄─┘
│ status      │  │    │ slot_number │  │    │ player_name │
│ created_at  │  └───►│ prize_name  │  └───►│ drawn_at    │
│ updated_at  │       │ prize_grade │       │ session_id  │
└─────────────┘       │ is_drawn    │       └─────────────┘
                      └─────────────┘
```

## 2. 테이블 상세 설계

### 2.1 games (게임 세션)
게임 전체를 관리하는 테이블

```sql
CREATE TABLE games (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,           -- 게임 이름 (예: "2024 송년회 뽑기")
    status VARCHAR(20) DEFAULT 'waiting', -- waiting, active, ended
    total_slots INTEGER DEFAULT 100,      -- 총 뽑기 칸 수
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 상태값
-- waiting: 대기중 (참가자 입장 불가)
-- active: 진행중 (뽑기 가능)
-- ended: 종료 (뽑기 불가, 결과만 조회)
```

### 2.2 prizes (경품/뽑기칸)
각 뽑기 칸의 정보를 저장

```sql
CREATE TABLE prizes (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    game_id UUID NOT NULL REFERENCES games(id) ON DELETE CASCADE,
    slot_number INTEGER NOT NULL,         -- 1~100 번호
    prize_name VARCHAR(200) NOT NULL,     -- 경품명 또는 "꽝"
    prize_grade VARCHAR(20),              -- 등급 (1등, 2등, ... 또는 null)
    prize_image_url TEXT,                 -- 경품 이미지 (선택)
    is_drawn BOOLEAN DEFAULT FALSE,       -- 뽑힘 여부
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),

    UNIQUE(game_id, slot_number)          -- 같은 게임 내 번호 중복 방지
);

-- 인덱스
CREATE INDEX idx_prizes_game_id ON prizes(game_id);
CREATE INDEX idx_prizes_is_drawn ON prizes(is_drawn);
```

### 2.3 draws (뽑기 기록)
참가자의 뽑기 결과 기록

```sql
CREATE TABLE draws (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    prize_id UUID NOT NULL REFERENCES prizes(id) ON DELETE CASCADE,
    game_id UUID NOT NULL REFERENCES games(id) ON DELETE CASCADE,
    player_name VARCHAR(50) NOT NULL,     -- 참가자 이름
    session_id VARCHAR(100),              -- 브라우저 세션 ID (중복 참여 방지용)
    drawn_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),

    UNIQUE(prize_id)                      -- 하나의 칸은 한 번만 뽑힘
);

-- 인덱스
CREATE INDEX idx_draws_game_id ON draws(game_id);
CREATE INDEX idx_draws_player_name ON draws(player_name);
CREATE INDEX idx_draws_session_id ON draws(session_id);
```

## 3. Row Level Security (RLS) 정책

### 3.1 games 테이블
```sql
-- 읽기: 모든 사용자 허용
CREATE POLICY "Games are viewable by everyone"
ON games FOR SELECT USING (true);

-- 쓰기: 관리자만 (서비스 역할 키 사용)
CREATE POLICY "Games are manageable by admin"
ON games FOR ALL USING (auth.role() = 'service_role');
```

### 3.2 prizes 테이블
```sql
-- 읽기: 모든 사용자 허용
CREATE POLICY "Prizes are viewable by everyone"
ON prizes FOR SELECT USING (true);

-- 쓰기: 관리자만
CREATE POLICY "Prizes are manageable by admin"
ON prizes FOR INSERT UPDATE DELETE USING (auth.role() = 'service_role');

-- is_drawn 업데이트: 함수를 통해서만
```

### 3.3 draws 테이블
```sql
-- 읽기: 모든 사용자 허용
CREATE POLICY "Draws are viewable by everyone"
ON draws FOR SELECT USING (true);

-- 쓰기: 함수를 통해서만 (anon 사용자도 가능)
CREATE POLICY "Draws can be created"
ON draws FOR INSERT WITH CHECK (true);
```

## 4. 핵심 함수 (Stored Procedures)

### 4.1 뽑기 실행 함수 (동시성 제어 포함)
```sql
CREATE OR REPLACE FUNCTION draw_prize(
    p_game_id UUID,
    p_slot_number INTEGER,
    p_player_name VARCHAR,
    p_session_id VARCHAR
)
RETURNS JSON
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
    v_prize RECORD;
    v_game_status VARCHAR;
    v_existing_draw RECORD;
BEGIN
    -- 1. 게임 상태 확인
    SELECT status INTO v_game_status
    FROM games WHERE id = p_game_id;

    IF v_game_status != 'active' THEN
        RETURN json_build_object(
            'success', false,
            'error', 'GAME_NOT_ACTIVE',
            'message', '게임이 진행중이 아닙니다.'
        );
    END IF;

    -- 2. 이미 참여했는지 확인 (선택적)
    SELECT * INTO v_existing_draw
    FROM draws
    WHERE game_id = p_game_id AND session_id = p_session_id;

    IF FOUND THEN
        RETURN json_build_object(
            'success', false,
            'error', 'ALREADY_PARTICIPATED',
            'message', '이미 참여하셨습니다.'
        );
    END IF;

    -- 3. 해당 칸 정보 조회 및 잠금 (FOR UPDATE)
    SELECT * INTO v_prize
    FROM prizes
    WHERE game_id = p_game_id AND slot_number = p_slot_number
    FOR UPDATE SKIP LOCKED;

    IF NOT FOUND THEN
        RETURN json_build_object(
            'success', false,
            'error', 'SLOT_NOT_FOUND',
            'message', '해당 번호를 찾을 수 없습니다.'
        );
    END IF;

    -- 4. 이미 뽑힌 칸인지 확인
    IF v_prize.is_drawn THEN
        RETURN json_build_object(
            'success', false,
            'error', 'ALREADY_DRAWN',
            'message', '이미 뽑힌 번호입니다.'
        );
    END IF;

    -- 5. 뽑기 처리
    UPDATE prizes SET is_drawn = true WHERE id = v_prize.id;

    INSERT INTO draws (prize_id, game_id, player_name, session_id)
    VALUES (v_prize.id, p_game_id, p_player_name, p_session_id);

    -- 6. 결과 반환
    RETURN json_build_object(
        'success', true,
        'prize_name', v_prize.prize_name,
        'prize_grade', v_prize.prize_grade,
        'slot_number', v_prize.slot_number
    );
END;
$$;
```

### 4.2 게임 초기화 함수
```sql
CREATE OR REPLACE FUNCTION reset_game(p_game_id UUID)
RETURNS VOID
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
    -- 뽑기 기록 삭제
    DELETE FROM draws WHERE game_id = p_game_id;

    -- 모든 칸 초기화
    UPDATE prizes SET is_drawn = false WHERE game_id = p_game_id;

    -- 게임 상태 대기로 변경
    UPDATE games SET status = 'waiting', updated_at = NOW()
    WHERE id = p_game_id;
END;
$$;
```
